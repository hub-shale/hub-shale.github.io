# Conduct SQL queries subject to the siting constraints for scenario 1 and merge queries

scenario_1_query = f"""SELECT {all_col}, ST_BUFFER(way, {H3}) as geom_way
FROM planet_osm_polygon
WHERE building in ('yes',
    'residential',
    'apartments',
    'house',
    'static_caravan',
    'detached')
    OR landuse = 'residential'
    OR place = 'town'
UNION
SELECT {all_col}, ST_BUFFER(way, {H3}) as geom_way
FROM planet_osm_polygon
WHERE building IS NOT NULL
UNION
SELECT {all_col}, ST_BUFFER(way, {km75}) as geom_way
    FROM planet_osm_polygon
    WHERE aeroway is not NULL
UNION
SELECT {all_col}, ST_BUFFER(way, 0) as geom_way
    FROM planet_osm_polygon
    WHERE landuse in ('military')
    OR military IS NOT NULL
UNION
SELECT {all_col}, ST_BUFFER(way, {H2}) as geom_way
    FROM planet_osm_line
    WHERE railway is not NULL
    OR railway in ('abandoned', 'razed', 'disused', 'dismantled')
UNION
SELECT {all_col}, ST_BUFFER(way, {H2}) as geom_way
    FROM planet_osm_line
    WHERE highway in ('motorway',
    'trunk', 
    'primary',
    'secondary',
    'motorway_link',
    'trunk_link',
    'primary_link',
    'secondary_link')
UNION
SELECT {all_col}, ST_BUFFER(way, 0) as geom_way
    FROM planet_osm_polygon as poly
    WHERE "natural" in ('wetland', 'yes')
    OR landuse in ('conservation')
    OR leisure in ('nature_reserve')
UNION
SELECT {all_col}, ST_BUFFER(way, {H1}) as geom_way
    FROM planet_osm_line
    WHERE waterway in ('river', 'stream', 'pond')
UNION
SELECT {all_col}, ST_BUFFER(way, 0) as geom_way
    FROM planet_osm_polygon
    WHERE water in ('lake')
UNION
SELECT {all_col}, ST_BUFFER(way, {H2}) as geom_way
    FROM planet_osm_line
    WHERE power IS NOT NULL
UNION
 SELECT {all_col}, ST_BUFFER(way, {H1}) as geom_way
    FROM planet_osm_polygon
    WHERE power IS NOT NULL
UNION
SELECT {all_col}, ST_BUFFER(way, {d5}) as geom_way
    FROM planet_osm_point
    WHERE "generator:source" IN ('wind')"""

  • Calculate the number of wind turbines that could be placed in each polygon

The turbine rotor diameter is 136 meters and the turbine towers must be at least 5 rotor diameters apart, therefore each turbine must be 680 meters apart from one another

In [39]:

  • Calculate the total annual energy production from the maximum possible number of new turbines

E = 2.6 s m-1 v + -5 GWh

E = energy production per turbine in GWh

v = average annual wind speed in m s-1
